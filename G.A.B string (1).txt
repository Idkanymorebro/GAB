-- Input box sec 2
Section2:NewTextBox("Head Size", "Increase the head of zombie(bombers doesn't work)", function(txt)
    _G.HeadSize = txt -- Adjust the head size as per your requirement
    _G.Disabled = false -- Set to true if you want the script initially disabled

    game:GetService('RunService').RenderStepped:Connect(function()
        if not _G.Disabled then
            local camera = workspace.CurrentCamera
            if camera then
                local zombies = camera:GetChildren() -- Get all children under Camera
                
                for _, zombie in ipairs(zombies) do
                    if zombie.Name == "m_Zombie" and zombie:IsA("Model") then
                        local head = zombie:FindFirstChild("Head") -- Find the Head part of the NPC

                        if head and head:IsA("MeshPart") then
                            pcall(function()
                                head.Size = Vector3.new(_G.HeadSize, _G.HeadSize, _G.HeadSize)
                                head.Transparency = 0.90
                                head.CanCollide = false
                            end)
                        end
                    end
                end
            end
        end
    end)
end)
