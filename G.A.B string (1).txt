-- Input box sec 2
Section2:NewTextBox("Head Size", "Increase the head size of zombies (bombers excluded)", function(headSize)
    _G.HeadSize = tonumber(headSize) or 2.5
    _G.Disabled = false

    game:GetService('RunService').RenderStepped:Connect(function()
        if not _G.Disabled then
            local camera = workspace.CurrentCamera
            if camera then
                for _, zombie in ipairs(camera:GetChildren()) do
                    if zombie.Name == "m_Zombie" and zombie:IsA("Model") then
                        local head = zombie:FindFirstChild("Head")
                        if head and head:IsA("MeshPart") then
                            pcall(function()
                                head.Size = Vector3.new(_G.HeadSize, _G.HeadSize, _G.HeadSize)
                                head.Transparency = 0.9
                                head.CanCollide = false
                            end)
                        end
                    end
                end
            end
        end
    end)
end)

-- For the torso
Section2:NewTextBox("Torso Size", "Adjust the torso size of zombies", function(torsoSize)
    _G.TorsoSize = tonumber(torsoSize) or 5
    _G.Disabled = false

    game:GetService('RunService').RenderStepped:Connect(function()
        if not _G.Disabled then
            local camera = workspace.CurrentCamera
            if camera then
                for _, zombie in ipairs(camera:GetChildren()) do
                    if zombie.Name == "m_Zombie" and zombie:IsA("Model") then
                        local ignoreModel = false
                        local ignoreList = {"Barrel", "Torch", "FTorso", "Whale oil lantern"}

                        for _, ignoreName in ipairs(ignoreList) do
                            if zombie:FindFirstChild(ignoreName, true) then
                                ignoreModel = true
                                break
                            end
                        end

                        if not ignoreModel then
                            local torso = zombie:FindFirstChild("Torso") -- Change made here
                            if torso and torso:IsA("Part") then
                                pcall(function()
                                    torso.Size = Vector3.new(_G.TorsoSize, _G.TorsoSize, _G.TorsoSize)
                                    torso.Transparency = 0.9
                                    torso.CanCollide = false

                                    for _, part in ipairs(zombie:GetDescendants()) do
                                        if part:IsA("Part") then
                                            part.Transparency = 0.9
                                        end
                                    end
                                end)
                            end
                        end
                    end
                end
            end
        end
    end)
end)
